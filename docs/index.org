(defn sqr [n] (* n n))

(sqr 10)
;=> 100

(sqr 0)
;=> 0

#+name: simple-example
#+begin_src clojure :tangle ../examples/simple.clj :noweb yes 
(contracts/provide 
  (sqr "Constraints for squaring" 
    [x] [number? (not= 0 x) => number? pos?]))
#+end_src
